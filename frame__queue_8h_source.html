104struct FrameQueue {
105 std::queue<AVFrame *> queue;
106 std::mutex lock;
107 AVFrame *eofFrame;
108
109 FrameQueue() : eofFrame(av_frame_alloc()) {}
110
111 ~FrameQueue() {
112 flush();
113 }
114
115 bool isEof(AVFrame *frame) {
116 return frame == eofFrame;
117 }
118
119 void pushEof() {
120 std::unique_lock<std::mutex> ul(lock);
121 queue.push(eofFrame);
122 }
123
124 void flush() {
125 std::unique_lock<std::mutex> ul(lock);
126 while (!queue.empty()) {
127 auto frame = queue.front();
128 if (frame != eofFrame)
129 av_frame_free(&frame);
130 queue.pop();
131 }
132 }
133
134 AVFrame *front() {
135 std::unique_lock<std::mutex> ul(lock);
136 if (queue.empty())
137 return nullptr;
138 return queue.front();
139 }
140
141 void pop() {
142 std::unique_lock<std::mutex> ul(lock);
143 if (queue.empty())
144 return;
145 queue.pop();
146 }
147
148 void push(AVFrame *frame) {
149 std::unique_lock<std::mutex> ul(lock);
150 queue.push(frame);
151 }
152};
